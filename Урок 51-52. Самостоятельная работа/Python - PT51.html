    [3, 4, 5],
    [12, 10, 12],
    [4, 5, 6, 9, 12, 12]
]

    avg_grade = sum(grade_list) / len(grade_list)
    if avg_grade >= 7:
        grade_list = True
    else:
        grade_list = False
