        if customers:
            self.__customers = customers
        else:
            self.__customers: list[Customer] = list()

        if admins:
            self.__admins = admins
        else:
            self.__admins: list[Admin] = list()

        if game_library:
            self.__games_library = game_library
        else:
            self.__games_library: list[GameProduct] = list()

    def __get_customer_index(self, login_to_find: str):
        found_index = -1
        customers = self.__customers
        for i in range(len(customers)):
            if customers[i].get_login() == login_to_find:
                found_index = i
                break
        return found_index

    def get_customer(self, login_to_find: str):
        found_index = self.__get_customer_index(login_to_find)
        found_customer = None
        if found_index != -1:
            found_customer = self.__customers[found_index]
        return found_customer

    def __get_admin_index(self, login_to_find: str):
        found_index = -1
        admins = self.__admins
        for i in range(len(admins)):
            if admins[i].get_login() == login_to_find:
                found_index = i
                break
        return found_index

    def get_admin(self, login_to_find: str):
        found_index = self.__get_admin_index(login_to_find)
        found_admin = None
        if found_index != -1:
            found_admin = self.__admins[found_index]
        return found_admin

        is_success = False
        found_customer = self.get_customer(customer_to_register.get_login())
        found_admin = self.get_admin(customer_to_register.get_login())
        if found_customer is None and found_admin is None:
            self.__customers.append(customer_to_register)
            is_success = True
        return is_success

        found_customer = self.get_customer(enter_login)

            is_success = True
        return is_success

При запуске программы пользователя встречает следующее меню: 

1. Войти
2. Зарегистрироваться
3. Завершить программу

При выборе пункта 2 программа запрашивает с консоли новый логин и пароль, затем пытается зарегистрировать нового пользователя. После этого программа сообщает о результате.

При выборе пункта 1 программа запрашивает с консоли логин и пароль, затем пытается провести вход пользователя. Если не удалось войти, то программа сообщает об этом. В противном случае, если логин и пароль от клиента, программа показывает в консоли главное меню клиента. Если же логин и пароль от администратора, то программа показывает в консоли главное меню администратора.

Главное меню клиента:

1. Мой аккаунт
2. Моя библиотека
3. Магазин
4. Выйти из аккаунта
5. Завершить программу

При выборе пункта 1 программа показывает меню аккаунта.

Меню аккаунта:

Логин: <название_логина>        Баланс: <число>
1. Пополнить баланс
2. Изменить логин
3. Изменить пароль
4. Вернуться

В меню аккаунта при выборе пункта 1 программа запрашивает с консоли количество денег и пытается пополнить баланс аккаунта. Затем программа сообщает результат.

В меню аккаунта при выборе пункта 2 программа запрашивает с консоли новый логин и пытается изменить его у пользователя. Затем программа сообщает результат.

В меню аккаунта при выборе пункта 3 программа запрашивает с консоли новый пароль и пытается изменить его у пользователя. Затем программа сообщает результат.

В главном меню клиента при выборе пункта 2 программа выводит в консоль все игры, которые есть на аккаунте.

В меню магазина при выборе пункта 5 программа запрашивает с консоли идентификатор. Если игра найдена то показывает меню изменения игры:

5. Отменить изменения и вернуться

В меню изменения игры при выборе пунктов 1-3 программа запрашивает соответствующие данные и меняет их у копии.

В меню изменения игры при выборе пункта 4 программа пытается подтвердить изменения. Если всё хорошо, то изменения фиксируются, об этом сообщается в консоли и программа возвращается в предыдущее меню. В противном случае программа сообщает о проблеме и остаётся в этом меню.
